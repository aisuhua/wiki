安装

```sh
shell> sudo apt-get install libboost-dev libboost-test-dev libboost-program-options-dev libboost-system-dev libboost-filesystem-dev libevent-dev automake libtool flex bison pkg-config g++ libssl-dev

shell> wget http://archive.apache.org/dist/thrift/0.9.3/thrift-0.9.3.tar.gz
shell> ./configure
shell> make
shell> make install

shell> thrift --version
Thrift version 0.9.3
```
